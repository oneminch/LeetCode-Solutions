The Problem: Single Number

My Solution:

def singleNumber(nums: List[int]) -> int:
    stack = []

    for n in nums:
        if n in stack:
            i = stack.index(n)
            stack.pop(i)
        else:
            stack.append(n)
        
    return stack[0]

The Problem: Reverse Integer

My Solution:

/**
 * @param {number} x
 * @return {number}
 */

var reverse = function(x) {
    var n = Number((Math.abs(x)).toString().split('').reverse().join(''));
    if (n > 0x7FFFFFFF) return 0;
    else return (x<0) ? -n : n;
};
